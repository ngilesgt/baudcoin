* perform gitian builds

  * From a directory containing the baudcoin source, gitian-builder and gitian.sigs
   export SIGNER=(your gitian key, ie bluematt, sipa, etc)
   export VERSION=0.5.1
   cd ./gitian-builder

  * Fetch and build inputs: (first time, or when dependency versions change)
   mkdir -p inputs; cd inputs/
   wget 'http://miniupnp.free.fr/files/download.php?file=miniupnpc-1.6.tar.gz' -O miniupnpc-1.6.tar.gz
   wget 'http://www.openssl.org/source/openssl-1.0.1c.tar.gz'
   wget 'http://download.oracle.com/berkeley-db/db-4.8.30.NC.tar.gz'
   wget 'http://zlib.net/zlib-1.2.6.tar.gz'
   wget 'ftp://ftp.simplesystems.org/pub/libpng/png/src/libpng-1.5.9.tar.gz'
   wget 'http://fukuchi.org/works/qrencode/qrencode-3.2.0.tar.bz2'
   wget 'http://downloads.sourceforge.net/project/boost/boost/1.50.0/boost_1_50_0.tar.bz2'
   wget 'http://releases.qt-project.org/qt4/source/qt-everywhere-opensource-src-4.8.2.tar.gz'
   cd ..
   ./bin/gbuild ../baudcoin/contrib/gitian-descriptors/boost-win32.yml
   mv build/out/boost-win32-1.50.0-gitian2.zip inputs/
   ./bin/gbuild ../baudcoin/contrib/gitian-descriptors/qt-win32.yml
   mv build/out/qt-win32-4.8.2-gitian-r1.zip inputs/
   ./bin/gbuild ../baudcoin/contrib/gitian-descriptors/deps-win32.yml
   mv build/out/baudcoin-deps-0.0.5.zip inputs/

  * Build baudcoind and baudcoin-qt on Linux32, Linux64, and Win32:
   ./bin/gbuild --commit baudcoin=v${VERSION} ../baudcoin/contrib/gitian-descriptors/gitian.yml
   ./bin/gsign --signer $SIGNER --release ${VERSION} --destination ../gitian.sigs/ ../baudcoin/contrib/gitian-descriptors/gitian.yml
   pushd build/out
   zip -r baudcoin-${VERSION}-linux-gitian.zip *
   mv baudcoin-${VERSION}-linux-gitian.zip ../../
   popd
   ./bin/gbuild --commit baudcoin=v${VERSION} ../baudcoin/contrib/gitian-descriptors/gitian-win32.yml
   ./bin/gsign --signer $SIGNER --release ${VERSION}-win32 --destination ../gitian.sigs/ ../baudcoin/contrib/gitian-descriptors/gitian-win32.yml
   pushd build/out
   zip -r baudcoin-${VERSION}-win32-gitian.zip *
   mv baudcoin-${VERSION}-win32-gitian.zip ../../
   popd

  Build output expected:
  1. linux 32-bit and 64-bit binaries + source (baudcoin-${VERSION}-linux-gitian.zip)
  2. windows 32-bit binary, installer + source (baudcoin-${VERSION}-win32-gitian.zip)
  3. Gitian signatures (in gitian.sigs/${VERSION}[-win32]/(your gitian key)/

* repackage gitian builds for release as stand-alone zip/tar/installer exe

  * Linux .tar.gz:
   unzip baudcoin-${VERSION}-linux-gitian.zip -d baudcoin-${VERSION}-linux
   tar czvf baudcoin-${VERSION}-linux.tar.gz baudcoin-${VERSION}-linux
   rm -rf baudcoin-${VERSION}-linux

  * Windows .zip and setup.exe:
   unzip baudcoin-${VERSION}-win32-gitian.zip -d baudcoin-${VERSION}-win32
   mv baudcoin-${VERSION}-win32/baudcoin-*-setup.exe .
   zip -r baudcoin-${VERSION}-win32.zip baudcoin-${VERSION}-win32
   rm -rf baudcoin-${VERSION}-win32

* perform Mac build
  See this blog post for how Gavin set up his build environment to build the OSX
  release; note that a patched version of macdeployqt is not needed anymore, as
  the required functionality and fixes are implemented directly in macdeployqtplus:
    http://gavintech.blogspot.com/2011/11/deploying-baudcoin-qt-on-osx.html
  Gavin also had trouble with the macports py27-appscript package; he
  ended up installing a version that worked with: /usr/bin/easy_install-2.7 appscript

  qmake RELEASE=1 USE_UPNP=1 USE_QRCODE=1 baudcoin-qt.pro
  make
  export QTDIR=/opt/local/share/qt4  # needed to find translations/qt_*.qm files
  T=$(contrib/qt_translations.py $QTDIR/translations src/qt/locale)
  python2.7 contrib/macdeploy/macdeployqtplus Baudcoin-Qt.app -add-qt-tr $T -dmg -fancy contrib/macdeploy/fancy.plist

 Build output expected:
  Baudcoin-Qt.dmg

* After 3 or more people have gitian-built, repackage gitian-signed zips:

  * From a directory containing baudcoin source, gitian.sigs and gitian zips
   export VERSION=0.5.1
   mkdir baudcoin-${VERSION}-linux-gitian
   pushd baudcoin-${VERSION}-linux-gitian
   unzip ../baudcoin-${VERSION}-linux-gitian.zip
   mkdir gitian
   cp ../baudcoin/contrib/gitian-downloader/*.pgp ./gitian/
   for signer in $(ls ../gitian.sigs/${VERSION}/); do
     cp ../gitian.sigs/${VERSION}/${signer}/baudcoin-build.assert ./gitian/${signer}-build.assert
     cp ../gitian.sigs/${VERSION}/${signer}/baudcoin-build.assert.sig ./gitian/${signer}-build.assert.sig
   done
   zip -r baudcoin-${VERSION}-linux-gitian.zip *
   cp baudcoin-${VERSION}-linux-gitian.zip ../
   popd
   mkdir baudcoin-${VERSION}-win32-gitian
   pushd baudcoin-${VERSION}-win32-gitian
   unzip ../baudcoin-${VERSION}-win32-gitian.zip
   mkdir gitian
   cp ../baudcoin/contrib/gitian-downloader/*.pgp ./gitian/
   for signer in $(ls ../gitian.sigs/${VERSION}-win32/); do
     cp ../gitian.sigs/${VERSION}-win32/${signer}/baudcoin-build.assert ./gitian/${signer}-build.assert

   done
   zip -r baudcoin-${VERSION}-win32-gitian.zip *
